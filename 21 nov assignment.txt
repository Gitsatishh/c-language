//18)  pallindrome
#include<stdio.h>
int main()
{
    int a;
    printf("Please enter your number: ");
    scanf("%d",&a);
    int temp=a;
    int rem,sum=0;
    while(a>0)
    {
        rem=a%10;
        sum=sum*10+rem;
        a=a/10;
    }
    if(temp==sum)
    {
        printf("The number is pallindrome");
    }
    else
    {
        printf("The number is not pallindrome");
    }
    return 0;
}
